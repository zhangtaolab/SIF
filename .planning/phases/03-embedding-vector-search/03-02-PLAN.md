    ```python
    class OpenAIEmbedder(Embedder):
        def __init__(
            self,
            model_name: str = "text-embedding-3-small",
            api_key: str | None = None,
            api_base: str | None = None,
            embedding_dim: int | None = None,
            cache_dir: str | None = None,
        ) -> None:
            try:
                from openai import OpenAI
            except ImportError:
                logger.error("openai not installed. Install with: pip install openai")
                raise
            self.model_name = model_name
            self._embedding_dim = embedding_dim
            self._client = OpenAI(api_key=api_key, base_url=api_base)
            self._cache_dir = Path(cache_dir) if cache_dir else None
            if self._cache_dir:
                self._cache_dir.mkdir(parents=True, exist_ok=True)
            self._dim_cache_file = self._cache_dir / "openai_dim_cache.json" if self._cache_dir else None
            if embedding_dim is None:
                self._embedding_dim = self._detect_dimension()

        def _detect_dimension(self) -> int:
            if self._dim_cache_file and self._dim_cache_file.exists():
                try:
                    import json
                    with open(self._dim_cache_file, "r", encoding="utf-8") as f:
                        cache = json.load(f)
                    if cache.get("model_name") == self.model_name:
                        return cache["dimension"]
                except Exception:
                    pass
            response = self._client.embeddings.create(input=["dimension probe"], model=self.model_name)
            dim = len(response.data[0].embedding)
            if self._dim_cache_file:
                try:
                    import json
                    with open(self._dim_cache_file, "w", encoding="utf-8") as f:
                        json.dump({"model_name": self.model_name, "dimension": dim}, f)
                except Exception:
                    pass
            return dim

        def embed(self, text: str) -> List[float]:
            response = self._client.embeddings.create(input=[text], model=self.model_name)
            return response.data[0].embedding

        def embed_batch(self, texts: List[str]) -> List[List[float]]:
            response = self._client.embeddings.create(input=texts, model=self.model_name)
            return [item.embedding for item in response.data]

        @property
        def dimension(self) -> int:
            return self._embedding_dim
    ```
