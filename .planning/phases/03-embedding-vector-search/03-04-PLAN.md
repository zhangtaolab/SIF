       ```python
       def add_embeddings_batch(
           self,
           items: List[tuple[str, str, Optional[str], List[float]]],
       ) -> None:
           """Add multiple embeddings in a single batch operation.

           Each item is a tuple of (embedding_id, document_id, chunk_id, embedding_vector).
           """
           if not items:
               return
           rows = [
               (eid, doc_id, chunk_id, json.dumps(vec))
               for eid, doc_id, chunk_id, vec in items
           ]
           self.db.executemany(
               """
               INSERT OR REPLACE INTO document_embeddings
               (embedding_id, document_id, chunk_id, embedding)
               VALUES (?, ?, ?, vec_f32(?))
               """,
               rows,
           )
       ```

    ```python
    import sqlite3
    import pytest
    from docsift.core.models import SearchOptions
    from docsift.search.vector import VectorSearcher

    def _load_vec(conn: sqlite3.Connection) -> bool:
        try:
            conn.enable_load_extension(True)
            conn.load_extension("vec0")
            return True
        except Exception:
            return False

    @pytest.fixture
    def vec_db():
        conn = sqlite3.connect(":memory:")
        conn.row_factory = sqlite3.Row
        if not _load_vec(conn):
            pytest.skip("sqlite-vec not available")
        conn.execute("""
            CREATE VIRTUAL TABLE document_embeddings USING vec0(
                embedding_id TEXT PRIMARY KEY,
                document_id TEXT NOT NULL,
                chunk_id TEXT,
                embedding FLOAT[384]
            )
        """)
        # Create minimal documents/collections tables for search joins
        conn.execute("""
            CREATE TABLE collections (
                id TEXT PRIMARY KEY,
                name TEXT NOT NULL
            )
        """)
        conn.execute("""
            CREATE TABLE documents (
                id TEXT PRIMARY KEY,
                collection_id TEXT NOT NULL,
                path TEXT NOT NULL,
                filename TEXT NOT NULL,
                title TEXT,
                content TEXT NOT NULL,
                checksum TEXT NOT NULL,
                file_size INTEGER DEFAULT 0,
                mtime REAL NOT NULL,
                created_at TEXT NOT NULL,
                updated_at TEXT NOT NULL,
                metadata TEXT DEFAULT '{}'
            )
        """)
        yield conn
        conn.close()

    class TestVectorSearcher:
        def test_init_fails_when_vec_unavailable(self):
            conn = sqlite3.connect(":memory:")
            conn.row_factory = sqlite3.Row
            with pytest.raises(RuntimeError, match="sqlite-vec extension is not available"):
                VectorSearcher(conn, embedding_dim=384)
            conn.close()

        def test_add_embedding_inserts_single_row(self, vec_db):
            searcher = VectorSearcher(vec_db, embedding_dim=384)
            embedding = [0.1] * 384
            searcher.add_embedding("e1", "d1", "c1", embedding)
            cursor = vec_db.execute("SELECT COUNT(*) FROM document_embeddings")
            assert cursor.fetchone()[0] == 1

        def test_add_embeddings_batch_inserts_multiple_rows(self, vec_db):
            searcher = VectorSearcher(vec_db, embedding_dim=384)
            items = [
                ("e1", "d1", "c1", [0.1] * 384),
                ("e2", "d1", "c2", [0.2] * 384),
                ("e3", "d2", None, [0.3] * 384),
            ]
            searcher.add_embeddings_batch(items)
            cursor = vec_db.execute("SELECT COUNT(*) FROM document_embeddings")
            assert cursor.fetchone()[0] == 3

        def test_add_embeddings_batch_empty_list_is_noop(self, vec_db):
            searcher = VectorSearcher(vec_db, embedding_dim=384)
            searcher.add_embeddings_batch([])
            cursor = vec_db.execute("SELECT COUNT(*) FROM document_embeddings")
            assert cursor.fetchone()[0] == 0

        def test_search_returns_results_ordered_by_distance(self, vec_db):
            searcher = VectorSearcher(vec_db, embedding_dim=384)
            # Insert a document and collection for join
            vec_db.execute("INSERT INTO collections (id, name) VALUES ('col1', 'notes')")
            vec_db.execute(
                "INSERT INTO documents (id, collection_id, path, filename, title, content, checksum, mtime, created_at, updated_at) "
                "VALUES ('d1', 'col1', '/a.md', 'a.md', 'A', 'content', 'chk', 1.0, '2024-01-01', '2024-01-01')"
            )
            searcher.add_embedding("e1", "d1", None, [0.5] * 384)
            query = [0.5] * 384
            options = SearchOptions(limit=5)
            results = searcher.search(query, options)
            assert len(results) == 1
            assert results[0].document_id == "d1"
            assert results[0].score > 0.0

        def test_search_with_collection_filter(self, vec_db):
            searcher = VectorSearcher(vec_db, embedding_dim=384)
            vec_db.execute("INSERT INTO collections (id, name) VALUES ('col1', 'notes'), ('col2', 'docs')")
            vec_db.execute(
                "INSERT INTO documents (id, collection_id, path, filename, title, content, checksum, mtime, created_at, updated_at) "
                "VALUES ('d1', 'col1', '/a.md', 'a.md', 'A', 'content', 'chk', 1.0, '2024-01-01', '2024-01-01')"
            )
            vec_db.execute(
                "INSERT INTO documents (id, collection_id, path, filename, title, content, checksum, mtime, created_at, updated_at) "
                "VALUES ('d2', 'col2', '/b.md', 'b.md', 'B', 'content', 'chk', 1.0, '2024-01-01', '2024-01-01')"
            )
            searcher.add_embeddings_batch([
                ("e1", "d1", None, [0.5] * 384),
                ("e2", "d2", None, [0.1] * 384),
            ])
            query = [0.5] * 384
            options = SearchOptions(limit=5, collection_ids=["col2"])
            results = searcher.search(query, options)
            assert len(results) == 1
            assert results[0].document_id == "d2"
    ```
