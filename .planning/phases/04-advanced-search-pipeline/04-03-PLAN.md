       ```python
       class SearchPipeline:
           def __init__(
               self,
               db: sqlite3.Connection,
               embedder: Optional[Embedder] = None,
               query_expander: Optional[QueryExpansion] = None,
               reranker: Optional[CrossEncoderReranker] = None,
               snippet_extractor: Optional[SmartSnippetExtractor] = None,
               embedding_dim: int = 768,
           ) -> None:
               self.db = db
               self.hybrid = HybridSearcher(db, embedder, embedding_dim)
               self.query_expander = query_expander
               self.reranker = reranker
               self.snippet_extractor = snippet_extractor or SmartSnippetExtractor()
           
           def search(
               self,
               query: str,
               options: Optional[SearchOptions] = None,
           ) -> List[SearchResult]:
               if options is None:
                   options = SearchOptions()
               
               # Parse query prefix
               parsed_query, search_type = self._parse_query_prefix(query)
               
               # Apply intent if provided
               if options.intent:
                   parsed_query = f"{options.intent}: {parsed_query}"
               
               # Route to appropriate search mode
               if search_type == SearchType.BM25:
                   return self.hybrid.bm25.search(parsed_query, options)
               elif search_type == SearchType.VECTOR:
                   if self.hybrid.embedder is None:
                       raise RuntimeError("Vector search requires an embedder")
                   query_embedding = self.hybrid.embedder.embed(parsed_query)
                   return self.hybrid.vector.search(query_embedding, options)
               elif search_type == SearchType.HYDE:
                   # Generate hypothetical document and search by vector
                   hyde_doc = self._generate_hypothetical_document(parsed_query)
                   if self.hybrid.embedder is None:
                       raise RuntimeError("HyDE search requires an embedder")
                   hyde_embedding = self.hybrid.embedder.embed(hyde_doc)
                   return self.hybrid.vector.search(hyde_embedding, options)
               
               # Default: hybrid search with optional expansion
               queries = [parsed_query]
               
               # Expand query if expander is available and prefix is expand: or default
               if self.query_expander is not None and search_type in (SearchType.HYBRID, SearchType.EXPAND):
                   try:
                       expanded = self.query_expander.expand(parsed_query)
                       # Deduplicate: keep original first, then unique expanded variants
                       seen = {parsed_query.lower()}
                       for variant in expanded[1:]:
                           if variant.lower() not in seen:
                               seen.add(variant.lower())
                               queries.append(variant)
                   except Exception as e:
                       logger.warning(f"Query expansion failed: {e}")
               
               # Search with each query
               all_results = []
               for q in queries:
                   results = self.hybrid.search(q, options)
                   if results:
                       all_results.append(results)
               
               # If only one query or no expansion, return results
               if len(all_results) <= 1:
                   results = all_results[0] if all_results else []
               else:
                   # Fuse results from multiple queries
                   results = self.hybrid.rrf.fuse(all_results, options.limit)
               
               # Apply candidate limit before reranking
               if self.reranker is not None and len(results) > 0:
                   candidates = results[:options.candidate_limit]
                   try:
                       reranked = self._rerank_with_explain(parsed_query, candidates, options)
                       results = reranked
                   except Exception as e:
                       # Fail fast at pipeline level per project conventions
                       raise RuntimeError(f"Reranking failed: {e}")
               
               # Extract smart snippets if not already present
               if self.snippet_extractor is not None:
                   for result in results:
                       if result.snippet is None and result.content:
                           query_terms = parsed_query.lower().split()
                           result.snippet = self.snippet_extractor.extract(result.content, query_terms)
               
               return results
           
           def _parse_query_prefix(self, query: str) -> tuple[str, SearchType]:
               """Parse query prefix to determine search mode."""
               PREFIX_MAP = {
                   "lex:": SearchType.BM25,
                   "vec:": SearchType.VECTOR,
                   "hyde:": SearchType.HYDE,
                   "expand:": SearchType.EXPAND,
               }
               for prefix, search_type in PREFIX_MAP.items():
                   if query.startswith(prefix):
                       return query[len(prefix):].strip(), search_type
               return query, SearchType.HYBRID
           
           def _generate_hypothetical_document(self, query: str) -> str:
               """Generate a hypothetical ideal answer document for HyDE search.
               
               For local-first operation, uses a simple prompt template with the embedding model
               if it supports text generation, otherwise returns the query as-is.
               """
               # Check if embedder has generate capability (GGUF models might)
               if hasattr(self.hybrid.embedder, 'generate'):
                   prompt = f"Answer this question concisely: {query}"
                   return self.hybrid.embedder.generate(prompt)
               # Fallback: use query as hypothetical document
               return query
           
           def _rerank_with_explain(
               self,
               query: str,
               candidates: List[SearchResult],
               options: SearchOptions,
           ) -> List[SearchResult]:
               """Rerank candidates and preserve scores for explainability."""
               # Build documents for reranking
               documents = []
               for result in candidates:
                   text = result.content or ""
                   if not text and result.highlights:
                       text = result.highlights[0]
                   documents.append(text)
               
               # Rerank
               reranked = self.reranker.rerank(query, candidates, top_k=options.limit)
               
               return reranked
       ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI -> SearchPipeline | Untrusted user query crosses into search execution |
| SearchPipeline -> BM25Searcher | Query enters FTS5 MATCH |
| SearchPipeline -> VectorSearcher | Embedding enters SQL parameter |
| SearchPipeline -> CrossEncoderReranker | Query + document text enter model inference |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-08 | Tampering | _parse_query_prefix | accept | Prefix stripping is simple string operation; no injection risk |
| T-04-09 | Tampering | _generate_hypothetical_document | mitigate | Uses embedding model generate() only if available; fallback to query string avoids untrusted generation |
| T-04-10 | Denial of Service | candidate_limit | mitigate | Default 20, CLI caps at 200; prevents excessive reranker latency |
| T-04-11 | Information Disclosure | --explain scores | accept | Local-first tool; internal scores are acceptable debug output |
| T-04-12 | Tampering | QueryExpansion with intent prefix | accept | Intent is prepended as plain string; no structured prompt injection risk |
</threat_model>
