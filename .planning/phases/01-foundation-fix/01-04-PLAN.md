    ```python
    def _create_gguf_model(
        self,
        model_path: str | None,
        model_name: str,
        **kwargs: dict[str, any],
    ) -> EmbeddingModel:
        """Create a GGUF model using llama-cpp-python."""
        from docsift.embedding.model import EmbeddingModel
        import numpy as np

        logger.info(f"Creating GGUF model: {model_name}")

        class GGUFModel(EmbeddingModel):
            def __init__(self, path: str | None, name: str, **kwargs):
                super().__init__(
                    model_id=name,
                    embedding_dim=kwargs.get("embedding_dim", 384),
                    max_tokens=kwargs.get("max_tokens", 512),
                )
                self._path = path
                self._n_gpu_layers = kwargs.get("n_gpu_layers", 0)
                self._n_ctx = kwargs.get("n_ctx", 2048)
                self._model: Any | None = None

            def load(self) -> None:
                from llama_cpp import Llama
                if self._path is None:
                    raise ValueError("model_path is required for GGUF models")
                self._model = Llama(
                    model_path=self._path,
                    n_ctx=self._n_ctx,
                    n_gpu_layers=self._n_gpu_layers,
                    embedding=True,
                    verbose=False,
                )
                self._loaded = True
                logger.info(f"Loaded GGUF model from {self._path}")

            def unload(self) -> None:
                self._model = None
                self._loaded = False

            def embed(
                self,
                texts: list[str],
                normalize: bool = True,
            ) -> list[list[float]]:
                if self._model is None:
                    raise RuntimeError("Model not loaded")
                embeddings = []
                for text in texts:
                    embedding = self._model.embed(text)
                    if normalize:
                        norm = np.linalg.norm(embedding)
                        if norm > 0:
                            embedding = embedding / norm
                    embeddings.append(embedding.tolist())
                return embeddings

        return GGUFModel(model_path, model_name, **kwargs)
    ```
